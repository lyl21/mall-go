CREATE TABLE IF NOT EXISTS optometry_records (    
    optometry_records_id bigint(20) AUTO_INCREMENT PRIMARY KEY COMMENT '验光记录ID',  
    create_time datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建日期',    
    update_time datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',   
    customer_id bigint(20) COMMENT '顾客ID',    
    customer_name VARCHAR(255) NOT NULL COMMENT '顾客姓名',    
    optometrist_id bigint(20) NOT NULL COMMENT '验光师ID',    
    optometrist_name VARCHAR(255) COMMENT '验光师姓名',    
    store_id bigint(20) NOT NULL COMMENT '门店ID',    
    store_name VARCHAR(255) NOT NULL COMMENT '门店名称',    
    store_phone VARCHAR(20) NOT NULL COMMENT '门店电话'  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='验光记录表';  
  
CREATE TABLE optometry_data (  
    optometry_id bigint(20) AUTO_INCREMENT PRIMARY KEY COMMENT '眼镜验光ID',  
    optometry_records_id bigint(20) COMMENT '验光记录ID',  
    create_time datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建日期',    
    update_time datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',    
    spherical_mirror VARCHAR(50) COMMENT '球镜数据',  
    cylindrical_mirror VARCHAR(50) COMMENT '柱镜数据',  
    position_of_axis VARCHAR(50) COMMENT '轴位数据',  
    addend VARCHAR(50) COMMENT 'ADD数据',  
    distance_of_pupil VARCHAR(50) COMMENT '瞳距数据',  
    horizontal_prism VARCHAR(50) COMMENT '水平棱镜数据',  
    vertical_prism VARCHAR(50) COMMENT '垂直棱镜数据',  
    type int(1) NOT NULL COMMENT '分类数据（1=电脑验光仪，2=查片仪，3=验光头，4=最终配镜）',  
    near_far int(1) COMMENT '远近数据（0=远，1=近）',  
    left_right_eyes int(1) NOT NULL COMMENT '左右眼数据（0=左眼，1=右眼）',  
    FOREIGN KEY (optometry_records_id) REFERENCES optometry_records(optometry_records_id) ON DELETE CASCADE ON UPDATE CASCADE  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='眼镜验光数据表';
